cmake_minimum_required(VERSION 3.22)
project(llama_jni_desktop LANGUAGES C CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# -------------------------------
# Paths
# -------------------------------
# This CMakeLists.txt lives at:
#   library/cmake/llama-jni-desktop/CMakeLists.txt
#
# Repo root is three levels up:
#   library/cmake/llama-jni-desktop -> library/cmake -> library -> repo root
get_filename_component(REPO_ROOT "${CMAKE_CURRENT_LIST_DIR}/../../.." ABSOLUTE)

set(LLAMA_CPP_DIR "${REPO_ROOT}/llama.cpp")
set(LLAMA_INCLUDE_DIR "${LLAMA_CPP_DIR}/include")

set(JNI_SRC "${REPO_ROOT}/library/src/commonMain/cpp/llama_jni.cpp")
set(JNI_INCLUDE_DIR "${REPO_ROOT}/library/src/commonMain/c_interop/include")

message(STATUS "REPO_ROOT        = ${REPO_ROOT}")
message(STATUS "LLAMA_CPP_DIR    = ${LLAMA_CPP_DIR}")
message(STATUS "LLAMA_INCLUDE_DIR= ${LLAMA_INCLUDE_DIR}")
message(STATUS "JNI_SRC          = ${JNI_SRC}")
message(STATUS "JNI_INCLUDE_DIR  = ${JNI_INCLUDE_DIR}")

if(NOT EXISTS "${JNI_SRC}")
    message(FATAL_ERROR "JNI source not found: ${JNI_SRC}")
endif()

if(NOT EXISTS "${LLAMA_CPP_DIR}/CMakeLists.txt")
    message(FATAL_ERROR "llama.cpp not found at: ${LLAMA_CPP_DIR}")
endif()

# -------------------------------
# Build llama.cpp from your repo (NOT system headers)
# -------------------------------
# Disable stuff you don't need for a shared JNI lib build
set(LLAMA_BUILD_TESTS OFF CACHE BOOL "" FORCE)
set(LLAMA_BUILD_EXAMPLES OFF CACHE BOOL "" FORCE)
set(LLAMA_BUILD_SERVER OFF CACHE BOOL "" FORCE)
set(LLAMA_CURL OFF CACHE BOOL "" FORCE)
set(GGML_OPENMP OFF CACHE BOOL "" FORCE)

# Build llama.cpp into a dedicated sub-build folder
add_subdirectory("${LLAMA_CPP_DIR}" "${CMAKE_BINARY_DIR}/llama-wrapper-build")

# -------------------------------
# JNI
# -------------------------------
find_package(JNI REQUIRED)

add_library(llama_jni SHARED "${JNI_SRC}")

# IMPORTANT:
# Put *repo* llama.cpp/include FIRST so "llama.h" resolves to the correct one
target_include_directories(llama_jni
        PRIVATE
        "${LLAMA_INCLUDE_DIR}"
        "${JNI_INCLUDE_DIR}"
        "${JNI_INCLUDE_DIR}/.."      # optional, depending on where llama_jni.h sits
        ${JNI_INCLUDE_DIRS}
)

# If you want to be extra-defensive against /usr/local/include picking the wrong header:
# target_compile_options(llama_jni PRIVATE -Wno-deprecated-declarations)

# Link against the llama.cpp targets we just built.
# In llama.cpp builds, the main library target is typically "llama".
# Your build log shows libllama.a being built, so target "llama" exists.
target_link_libraries(llama_jni
        PRIVATE
        llama
)

# macOS convenience: make sure the output is libllama_jni.dylib
set_target_properties(llama_jni PROPERTIES
        OUTPUT_NAME "llama_jni"
)
